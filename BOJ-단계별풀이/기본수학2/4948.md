while (true) {
    let n: Int = Int(readLine()!)!

    if n == 0 { break }

    var primeNums: [Int] = Array(repeating: 0, count: 2*n+1)
    var count: Int = 0

    for i in 2...(2*n) {
        primeNums[i] = i
    }
    for i in 2...(2*n) {
        if primeNums[i] == 0 { continue }
        for j in stride(from: i+i, through: 2*n, by: i) {
            primeNums[j] = 0
        }
    }
    for i in (n+1)...(2*n) {
        if primeNums[i] != 0 {
            count += 1
        }
    }
    print(count)
}
