```Swift
var nums: [Int] = []
var originNums: [Int] = Array(1...10000)

var sum: Int = 0
var stringNum: String = ""

func d(_ n: Int) {
    sum += n
    stringNum = String(n)

    for num in stringNum {
        sum += Int(String(num))!
    }

    nums.append(sum)
    sum = 0
}

for n in 1...10000 {
    d(n)
}

let selfNums = originNums.filter { !nums.contains($0) }

for num in selfNums {
    print(num)
}
```
