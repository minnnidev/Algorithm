let input: [Int] = readLine()!.split(separator: " ").map { Int($0)! }
let M: Int = input[0]
let N: Int = input[1]
var primeNums: [Int] = Array(repeating: 0, count: N+1)

for i in 2...N {
    primeNums[i] = i
}

for i in 2...N {
    if primeNums[i] == 0 { continue }
    for j in stride(from: i+i, through: N, by: i) {
        primeNums[j] = 0
    }
}

for i in M...N {
    if primeNums[i] != 0 {
        print(primeNums[i])
    }
}
