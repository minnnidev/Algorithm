let S: String = readLine()!
var word: String = ""
var tag: String = ""

for char in S {
    if char == "<" {
        print(String(word.reversed()), terminator: "")
        word = ""
        tag.append("<")
    } else if char == ">" {
        tag.append(">")
        print(tag, terminator: "")
        tag = ""
    } else {
        if tag.first == "<" {
            tag.append(char)
        } else if char == " " {
            print(String(word.reversed()), terminator: " ")
            word = ""
        } else {
            word.append(char)
        }
    }
}

if !word.isEmpty {
    print(String(word.reversed()))
}
