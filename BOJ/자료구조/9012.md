struct Stack<T> {
    private var stack: [T] = []
    var size: Int {
        return stack.count
    }
    var isEmpty: Bool {
        return stack.isEmpty
    }
    var top: T? {
        return isEmpty ? nil : stack[size-1]
    }
    mutating func push(_ element: T) {
        stack.append(element)
    }
    mutating func pop() -> T? {
        return isEmpty ? nil : stack.popLast()
    }
}

let T: Int = Int(readLine()!)!
var isVPS: Bool = true

for _ in 0..<T {
    let line: String = readLine()!
    var stack: Stack = Stack<Character>()
    isVPS = true

    for word in line {
        if word == "(" {
            stack.push(word)
        } else if word == ")" {
            if stack.size == 0 {
                isVPS = false
                break
            } else {
                _ = stack.pop()
            }
        }
    }
    print(isVPS && stack.isEmpty ? "YES" : "NO")
}
