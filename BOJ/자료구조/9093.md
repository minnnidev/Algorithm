    var size: Int {
        return stack.count
    }
    var isEmpty: Bool {
        return stack.isEmpty
    }
    var top: T? {

    }
    mutating func pop() -> T? {
        return isEmpty ? nil : stack.popLast()
    }
}

for _ in 0..<T {

        for j in word {
            stack.push(j)
        }

            print(stack.pop()!, terminator: "")
        }
        print(" ", terminator: "")
    }
    print()
}
